model chromosomeSegregation

  # inputs (modeled as parameters for now)
  ChromosomeIsSegregated          = 0; 
  Chromosome_polymerizedRegions   = 232304;
  Chromosome_supercoiled          = 4;


  # input enzymes
  species CobQ          = 1;             # MG_470_MONOMER           # CobQ/CobB/MinD/ParA nucleotide binding domain
  species mraZ          = 1;             # MG_221_OCTAMER           # mraZ protein
  species Era           = 1;             # MG_387_MONOMER           # GTP-binding protein Era
  species Obg           = 1;             # MG_384_MONOMER           # GTPase1 Obg
  species topoisomerase = 1;             # MG_203_204_TETRAMER      # DNA topoisomerase IV


  # input metabolites
  species gtp           = 500;
  species gdp           = 500;
  species h             = 500;
  species h2o           = 4;
  species p             = 500;

  # dummy reaction to test the system
  #=> h2o; 1;
  
  # parameters
  Chromosome_nCompartments  = 4; # this should come directly from the chromosome class
  Chromosome_sequenceLen    = 58076; # value taken from knowlege base 
  gtpCost                   = 1; # TODO check this value in the knowlege base


  # the propagation-logic is encoded as an event
  # if conditions are fulfilled segregation is executed
  TriggerChromosomeSegregation: at (h2o >= gtpCost \
     				and Chromosome_polymerizedRegions == Chromosome_nCompartments * Chromosome_sequenceLen \   
     				and Chromosome_supercoiled == Chromosome_nCompartments \                                   
     				and CobQ >= 1 \
				and mraZ >= 1 \
     				and Era >= 1 \
     				and Obg >= 1 \
     				and topoisomerase >= 1 \
     				and gtp >= gtpCost \
     				and h2o >= gtpCost), \
				t0=false: \
     				h2o = h2o - gtpCost, \
     				gtp = gtp - gtpCost, \ 
     				gdp = gdp + gtpCost, \
    				p   = p   + gtpCost, \
     				h   = h   + gtpCost, \
     				ChromosomeIsSegregated = 1;
end
